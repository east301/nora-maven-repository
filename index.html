Usage - Gradle

repositories {
    maven {
        url 'https://east301.github.io/nora-maven-repository'
    }
}

dependencies {
    compile 'jhdf5:jhdf5:14.12.2'
    compile 'org.openscience.cdk:cdk-annotation:1.5.8'
    ...
}
